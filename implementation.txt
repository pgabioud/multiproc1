#include <iostream>
#include <iomanip>
#include <sys/time.h>
#include <cuda_runtime.h>
using namespace std;

// CPU Baseline
void array_process(double *input, double *output, int length, int iterations)
{
    double *temp;

    for(int n=0; n<(int) iterations; n++)
    {
        for(int i=1; i<length-1; i++)
        {
            for(int j=1; j<length-1; j++)
            {
                output[(i)*(length)+(j)] = (input[(i-1)*(length)+(j-1)] +
                                            input[(i-1)*(length)+(j)]   +
                                            input[(i-1)*(length)+(j+1)] +
                                            input[(i)*(length)+(j-1)]   +
                                            input[(i)*(length)+(j)]     +
                                            input[(i)*(length)+(j+1)]   +
                                            input[(i+1)*(length)+(j-1)] +
                                            input[(i+1)*(length)+(j)]   +
                                            input[(i+1)*(length)+(j+1)] ) / 9;

            }
        }
        output[(length/2-1)*length+(length/2-1)] = 1000;
        output[(length/2)*length+(length/2-1)]   = 1000;
        output[(length/2-1)*length+(length/2)]   = 1000;
        output[(length/2)*length+(length/2)]     = 1000;

        temp = input;
        input = output;
        output = temp;
    }
}


// GPU Optimized function
__global__
void GPU_array_process(double *input, double *output, int length, int iterations)
{
    //Cuda events for calculating elapsed time
    cudaEvent_t cpy_H2D_start, cpy_H2D_end, comp_start, comp_end, cpy_D2H_start, cpy_D2H_end;
    cudaEventCreate(&cpy_H2D_start);
    cudaEventCreate(&cpy_H2D_end);
    cudaEventCreate(&cpy_D2H_start);
    cudaEventCreate(&cpy_D2H_end);
    cudaEventCreate(&comp_start);
    cudaEventCreate(&comp_end);

    /* Preprocessing goes here */
    double *gpu_input;
    double *gpu_output;
    double *gpu_temp;
    
    cudaMalloc( (void**)&gpu_input, length);
    cudaMalloc( (void**)&gpu_output, length);

    cudaEventRecord(cpy_H2D_start);
    /* Copying array from host to device goes here */
    cudaMemcpy((void*)gpu_input, (void*)input, length, cudaMemcpyHostToDevice);
    cudaMemcpy((void*)gpu_output, (void*)output, length, cudaMemcpyHostToDevice);
    
    cudaEventRecord(cpy_H2D_end);
    cudaEventSynchronize(cpy_H2D_end);

    //Copy array from host to device
    cudaEventRecord(comp_start);
    /* GPU calculation goes here */
    

    for(int n=0; n<(int) iterations; n++)
    {
        for(int i=1; i<length-1; i++)
        {
            for(int j=1; j<length-1; j++)
            {
                gpu_output[(i)*(length)+(j)] = (gpu_input[(i-1)*(length)+(j-1)] +
                                            gpu_input[(i-1)*(length)+(j)]   +
                                            gpu_input[(i-1)*(length)+(j+1)] +
                                            gpu_input[(i)*(length)+(j-1)]   +
                                            gpu_input[(i)*(length)+(j)]     +
                                            gpu_input[(i)*(length)+(j+1)]   +
                                            gpu_input[(i+1)*(length)+(j-1)] +
                                            gpu_input[(i+1)*(length)+(j)]   +
                                            gpu_input[(i+1)*(length)+(j+1)] ) / 9;

            }
        }
        gpu_output[(length/2-1)*length+(length/2-1)] = 1000;
        gpu_output[(length/2)*length+(length/2-1)]   = 1000;
        gpu_output[(length/2-1)*length+(length/2)]   = 1000;
        gpu_output[(length/2)*length+(length/2)]     = 1000;

        gpu_temp = gpu_input;
        gpu_input = gpu_output;
        gpu_output = gpu_temp;
    }
    
    
    cudaEventRecord(comp_end);
    cudaEventSynchronize(comp_end);

    cudaEventRecord(cpy_D2H_start);
    /* Copying array from device to host goes here */
    cudaMemcpy((void*)input, (void*)gpu_input, length, cudaMemcpyDeviceToHost);
    cudaMemcpy((void*)output, (void*)gpu_output, length, cudaMemcpyDeviceToHost);
    
    cudaEventRecord(cpy_D2H_end);
    cudaEventSynchronize(cpy_D2H_end);

    free()

    float time;
    cudaEventElapsedTime(&time, cpy_H2D_start, cpy_H2D_end);
    cout<<"Host to Device MemCpy takes "<<setprecision(4)<<time/1000<<"s"<<endl;

    cudaEventElapsedTime(&time, comp_start, comp_end);
    cout<<"Computation takes "<<setprecision(4)<<time/1000<<"s"<<endl;

    cudaEventElapsedTime(&time, cpy_D2H_start, cpy_D2H_end);
    cout<<"Device to Host MemCpy takes "<<setprecision(4)<<time/1000<<"s"<<endl;
}
